contextmenu()

Bắt sự kiện khi người dùng bấm chuột phải trên một phần tử

            $("button").contextmenu(function(event) {
                console.log("Context menu triggered on button");
                event.preventDefault(); // Ngăn chặn hiển thị menu mặc định của trình duyệt
            });

            $(document).on("contextmenu", function(event) {
                console.log("Key pressed: " + event.key);
            });
