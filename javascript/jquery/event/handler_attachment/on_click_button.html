Mặc định type của button là type='submit' do đó ta cần dùng type='button' hoặc event.preventDefault() để tránh việc tải lại hoặc chuyển hướng trang. 
            <form>
                <input type="file" id="file-upload" name="file-upload"/>
                <button type="button">Upload</button>
            </form>

            $("form").on("submit", function(event) {
                event.preventDefault();  // Ngăn submit form nhưng vẫn cho phép sự kiện lan truyền
            });
