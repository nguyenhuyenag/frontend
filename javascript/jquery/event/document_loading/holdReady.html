HoldReady

Phương thức holdReady() được sử dụng để trì hoãn việc thực thi sự kiện document.ready cho đến khi mã lệnh gọi jQuery.ready(true)

Ví dụ

            // Giữ lại sự kiện ready
            jQuery.holdReady(true);

            // Giả lập một tác vụ không đồng bộ mất một khoảng thời gian để hoàn tất
            setTimeout(function() {
                console.log("Tác vụ không đồng bộ đã hoàn tất!");
                // Phóng thích sự kiện ready sau khi tác vụ không đồng bộ hoàn tất
                jQuery.holdReady(false);
            }, 3000);

            // Mã lệnh sau sẽ chạy khi tài liệu sẵn sàng
            $(document).ready(function() {
                console.log("Tài liệu đã sẵn sàng!");
            });

Ví dụ

            $(document).ready(function() {
                $.holdReady(true);
                // Load the additional script
                $.getScript("script.js", function() {
                  // Now that the script has loaded, execute our code
                  alert("The script has loaded!");
              
                  // Release the hold on the ready event
                  $.holdReady(false);
                });
              });
