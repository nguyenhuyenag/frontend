Callback là gì?

Callback là một hàm sẽ được thực hiện sau khi một hàm khác đã thực hiện xong

Ví dụ 1

            function first(){
                console.log("Một");
            }
            function second(){
                console.log("Hai");
            }
            first();
            second();
Kết quả sẽ là 
            Một
            Hai

Ví dụ 2

            function first() {
                setTimeout(function() {
                    console.log("Một");
                }, 1000);
            }
            function second() {
                console.log("Hai");
            }
Kết quả sẽ là 
            Hai
            Một

Do đó ta cần callback để đảm bảo code nhất định không thực thi cho đến khi code khác thực hiện xong.

Callback 1

            function doHomework(subject, callback) {
                console.log(`Bắt đầu làm bài tập ${subject}.`);
                callback();
            }
            doHomework('TOÁN', function() {
                console.log('Làm bài tập xong!');
            });

Callback 2

            function doHomework(subject, callback) {
                setTimeout(function() {
                    console.log(`Bắt đầu làm bài tập ${subject}.`);
                    callback();
                }, 1000);
            }
            function alertFinished() {
                console.log('Làm bài tập xong!');
            }
            doHomework('TOÁN', alertFinished);
